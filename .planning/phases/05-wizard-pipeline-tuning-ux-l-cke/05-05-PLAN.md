    ```typescript
    export function buildSectionPrompt(
      facts: WizardFacts,
      abschnitt: AntragsAbschnitt,
      richtlinie: Richtlinie | null,
      // ...
    ): string {
      const basePrompt = /* existierender Prompt-Bau */;
      if (!PIPELINE_CONFIG.useVorbildFormulierungen || !richtlinie) {
        return basePrompt;
      }

      const vorbilder = (richtlinie.vorbildFormulierungen ?? [])
        .filter(v => !v.abschnitt_id || v.abschnitt_id === abschnitt.id);
      const bestPractices = richtlinie.bestPractices ?? [];
      const rejectGruende = richtlinie.rejectGruende ?? [];

      if (vorbilder.length === 0 && bestPractices.length === 0 && rejectGruende.length === 0) {
        return basePrompt;
      }

      const injectionBlock = [
        vorbilder.length > 0 && `## Vorbild-Formulierungen für "${abschnitt.name}"`,
        vorbilder.length > 0 && vorbilder.map(v => `- "${v.text}"`).join("\n"),
        bestPractices.length > 0 && `\n## Best Practices erfolgreicher Anträge`,
        bestPractices.length > 0 && bestPractices.slice(0, 3).map(b => `- ${b.text}`).join("\n"),
        rejectGruende.length > 0 && `\n## Vermeide diese Reject-Muster`,
        rejectGruende.length > 0 && rejectGruende.slice(0, 3).map(r => `- ${r.text}`).join("\n"),
      ].filter(Boolean).join("\n");

      return `${basePrompt}\n\n${injectionBlock}`;
    }
    ```
