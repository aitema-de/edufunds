**Slot-Coverage-Funktion (D-14, Soft-Match per Wort-Vorkommen):**
```typescript
function computeSlotCoverage(frage: string, expectedSlots: string[]): number {
  if (expectedSlots.length === 0) return 1.0;
  const lower = frage.toLowerCase();
  const keywords: Record<string, string[]> = {
    bundesland: ["bundesland", "land", "region", "wo ", "welchem land", "berlin", "bayern", "nrw", "hessen", "sachsen"],
    zielgruppe: ["zielgruppe", "schultyp", "klasse", "klassenstufe", "wer", "schueler", "kinder", "grundschule", "gymnasium"],
    thema:      ["thema", "schwerpunkt", "fokus", "bereich", "worum", "was genau", "welches projekt", "art"],
  };
  let hits = 0;
  for (const slot of expectedSlots) {
    const kws = keywords[slot] ?? [slot];
    if (kws.some(kw => lower.includes(kw))) hits++;
  }
  return hits / expectedSlots.length;
}
```

const start = Date.now();
let result: MatchResult | null = null;
try {
  result = await runMatch(input);
} catch (err) {
  errMsg = String(err instanceof Error ? err.message : err);
  console.warn(`${LOG_PREFIX} Eintrag ${entry.id} fehlgeschlagen:`, errMsg);

// Tagged Union dispatchen:
const expectedClarif = entry.expected_clarification ?? false;
let clarifResult: EntryResult["clarifResult"] = "not_applicable";
let slotCoverage: number | null = null;
let actualTop3: EntryResult["actual_top3"] = [];
let recall: number | null = null;
let offTargetHit: boolean | null = null;
let totalCandidates = 0;
let filteredOut = 0;
let costs = { eurCents: 0, usdCents: 0, calls: 0, totalTokens: 0 };

if (result) {
  if (result.kind === "clarification") {
    // Slot-Coverage messen
    if (entry.expected_missing_slots && entry.expected_missing_slots.length > 0) {
      slotCoverage = computeSlotCoverage(result.question, entry.expected_missing_slots);
    }
    // clarifResult bestimmen
    clarifResult = expectedClarif ? "hit" : "false_pos";
    costs = { eurCents: result.costs.eurCents, usdCents: result.costs.usdCents, calls: result.costs.calls, totalTokens: result.costs.totalTokens };
  } else { // result.kind === "ranking"
    actualTop3 = result.matches.map((m) => ({
      id: m.id, score: m.score, passt_weil: m.passt_weil, achtung_bei: m.achtung_bei,
    }));
    // Recall + Off-Target wie bisher (auf entry.expected_top3/off_target)
    if (entry.expected_top3.length > 0) {
      const overlap = entry.expected_top3.filter(id => actualTop3.some(a => a.id === id)).length;
      recall = overlap / entry.expected_top3.length;
    }
    offTargetHit = actualTop3.some(a => entry.expected_off_target.includes(a.id));
    totalCandidates = result.totalCandidates;
    filteredOut = result.filteredOut;
    costs = { eurCents: result.costs.eurCents, usdCents: result.costs.usdCents, calls: result.costs.calls, totalTokens: result.costs.totalTokens };
    // clarifResult bestimmen
    clarifResult = expectedClarif ? "miss" : (entry.expected_clarification === false || entry.expected_clarification === undefined ? "not_applicable" : "not_applicable");
    // Vereinfachung: wenn expected=true UND actual=ranking → miss; sonst not_applicable
  }
